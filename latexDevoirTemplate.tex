\begin{homeworkProblem}
\begin{center}
    \textit{Donnez un exemple de langage fini et un autre de langage infini}
\end{center}
Soit l'alphabet $\sum= \{a\}$.
\begin{center}
Langage fini: $L = \{a\}$\\ 
Langage infini: $L_2 = \Sigma^*=  \{a^n\big| n \in \naturel\}$
\end{center}
Le premier a trivialement une cardinalité finie et pour le second, on a $|L_2| = |\naturel| =\infty$ comme voulu.  

\end{homeworkProblem}
\begin{homeworkProblem}

\begin{center}
    \textit{Est-ce que $(L^*)^*=L^*$?}
\end{center}


\textit{Note}: L'étoile de Kleene transforme $L$ en monoïde sous l'opération de la concaténation de mots ( $\e$ est le neutre et on a la fermeture par définition de $L^*$)\\

La solution a été retirée pour éviter que le document circule et gâche la question pour des étudiants qui feront le cours dans les prochaines années.
\end{homeworkProblem}

\begin{homeworkProblem}
\textbf{Exemples de notation:}
\begin{gather*}
    L^* = \bigcup_{k\in \naturel} L^k\\
    \Sigma = \{a,b,c\}\\
    w = a_1a_2\dots a_n \qquad \qquad \forall i \in [n],  a_i \in \Sigma 
\end{gather*}

\textbf{Exemple de preuve}\\
Montrons que \[\sum_{k=1}^n = \frac{n(n+1)}{2}\]
\begin{proof}\hfill \break
\textit{par induction}\hfill \break

\underline{cas de base} : $k=1$
\[\frac{1(1+1)}{2} = \frac{2}{2} = 1 \]

\underline{pas inductif}
\begin{gather*}
    \sum_{k=1}^{n+1} = \sum_{k=1}^{n} + n+1\\
    \xRightarrow{H.I} = \frac{n(n+1)}{2} + n + 1\\
    \intertext{je vais sauter l'arithmétique}
    = \frac{(n+1)(n+2)}{2}
\end{gather*}
Comme voulu\\
Par induction, on a bien que $\frac{n(n+1)}{2}$


\end{proof}

\end{homeworkProblem}
